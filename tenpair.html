    <script>

        let numbers = [];

        let selectedIndex = null;


        // Iniciar con una secuencia aleatoria de números

        function initGame() {

            numbers = [];

            for (let i = 0; i < 27; i++) {

                numbers.push({ val: Math.floor(Math.random() * 9) + 1, status: 'active' });

            }

            render();

        }


        function render() {

            const grid = document.getElementById('grid');

            grid.innerHTML = '';

            numbers.forEach((num, index) => {

                const div = document.createElement('div');

                div.className = 'cell';

                if (num.status === 'cleared') div.classList.add('cleared');

                if (selectedIndex === index) div.classList.add('selected');

                

                div.textContent = num.val;

                div.onclick = () => handleSelect(index);

                grid.appendChild(div);

            });

            checkWin();

        }


        function handleSelect(index) {

            if (numbers[index].status === 'cleared') return;


            if (selectedIndex === null) {

                selectedIndex = index;

            } else if (selectedIndex === index) {

                selectedIndex = null;

            } else {

                if (canMatch(selectedIndex, index)) {

                    numbers[selectedIndex].status = 'cleared';

                    numbers[index].status = 'cleared';

                    selectedIndex = null;

                } else {

                    selectedIndex = index; // Cambiar selección al nuevo

                }

            }

            render();

        }


        function canMatch(idx1, idx2) {

            const val1 = numbers[idx1].val;

            const val2 = numbers[idx2].val;


            // Regla: Iguales o suman 10

            if (val1 === val2 || val1 + val2 === 10) {

                return areAdjacent(idx1, idx2);

            }

            return false;

        }


        function areAdjacent(idx1, idx2) {

            let start = Math.min(idx1, idx2);

            let end = Math.max(idx1, idx2);


            // Verificar si están seguidos ignorando los "cleared"

            for (let i = start + 1; i < end; i++) {

                if (numbers[i].status === 'active') return false;

            }

            return true;

        }


        function addNumbers() {

            const currentActive = numbers.filter(n => n.status === 'active');

            const newBatch = currentActive.map(n => ({ val: n.val, status: 'active' }));

            numbers = [...numbers, ...newBatch];

            render();

        }


        function checkWin() {

            if (numbers.every(n => n.status === 'cleared')) {

                document.getElementById('message').innerText = "¡FELICIDADES! HAS GANADO";

            }

        }


        function resetGame() {

            document.getElementById('message').innerText = "";

            selectedIndex = null;

            initGame();

        }


        initGame();
